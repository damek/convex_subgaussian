\chapter{Module guide}\label{chap:modules}

This chapter records the role of each Lean file.

\section{Theorem statements}

\begin{itemize}
\item \textbf{\texttt{ConvexSubgaussian/GaussianMain.lean}}:
  the main theorem file. Defines \texttt{AdmissibleScale}, \texttt{cStar},
  and the sharp comparison theorem
  \texttt{sharp\_convexSubgaussianComparison}. Start here.
\item \textbf{\texttt{ConvexSubgaussian/GaussianAsymptotics.lean}}:
  exposes the explicit constants ($t_0, A, B, a, p_0, z, c_0$) and the
  two named results \texttt{gaussianDomination\_c0} and
  \texttt{gaussianSharpness\_below\_c0}.
\item \textbf{\texttt{ConvexSubgaussian/CoreDefs.lean}}:
  basic definitions: the hinge function, stop-loss transform, and the
  sub-Gaussian tail condition.
\item \textbf{\texttt{ConvexSubgaussian.lean}}:
  umbrella import for the package.
\end{itemize}

\section{Proof internals}

\begin{itemize}
\item \textbf{\texttt{ConvexSubgaussian/Internal/GaussianProof.lean}}:
  the analytic proof body. Contains the layer-cake lemmas,
  envelope construction, Gaussian closed-form calculations, monotone
  ratio argument, construction of the extremal measure, and the reduction
  from stop-loss comparison to convex domination.
\end{itemize}

\section{Suggested reading order}

\begin{enumerate}
\item \texttt{GaussianMain.lean} for the theorem statement.
\item \texttt{GaussianAsymptotics.lean} for the explicit constants
  and the domination/sharpness split.
\item This blueprint to navigate the proof in
  \texttt{Internal/GaussianProof.lean}.
\end{enumerate}
